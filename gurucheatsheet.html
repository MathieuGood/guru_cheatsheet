Lambda functions

	
>>> def add(x, y):
	return x + y

>>> add(5, 3)
8

# Is equivalent to the lambda function:
>>> add = lambda x, y: x + y
>>> add(5, 3)
8

# It's not even need to bind it to a name like add before:
>>> (lambda x, y: x + y)(5, 3)
8

# Justifying Text with rjust(), ljust(), and center()
rjust() and ljust():

>>> 'Hello World'.rjust(20)
'         Hello World'
>>> 'Hello'.ljust(10)
'Hello     '

>>> 'Hello'.rjust(20, '*')
'***************Hello'

# Removing Whitespace with strip(), rstrip(), and lstrip()
>>> spam = '    Hello World     '
>>> spam.strip()
'Hello World'

>>> spam.lstrip()
'Hello World '

>>> spam.rstrip()
'    Hello World'

>>> spam = 'SpamSpamBaconSpamEggsSpamSpam'
>>> spam.strip('ampS')
'BaconSpamEggs'
